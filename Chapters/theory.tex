\begin{equation}\label{eq:eft_expansion}
    \mathcal{L}_{\rm{EFT}} = \mathcal{L}_{\rm{SM}} + \sum^{N_{\rm{dim6}}}_p \frac{w^{(6)}_p}{\Lambda^2}\mathcal{O}^{(6)}_p + \sum^{N_{\rm{dim8}}}_p \frac{w^{(8)}_p}{\Lambda^4}\mathcal{O}^{(8)}_p + ... ,
\end{equation}
